# API配置示例
# 复制此文件为 .env 并填入真实的API密钥

# SerpAPI密钥 - 用于Google搜索API
# 注册地址：https://serpapi.com/
SERPAPI_KEY=your_serpapi_key_here

# Google Custom Search API配置
# 获取方法：https://developers.google.com/custom-search/v1/introduction
GOOGLE_API_KEY=your_google_api_key_here
GOOGLE_CX=your_google_cx_here

# 使用方法：
# 1. 将此文件重命名为 .env
# 2. 填入真实的API密钥
# 3. 运行 python advanced_crawler.py

# 注意：即使不配置API，爬虫仍然可以正常工作
